A motorized fan fades between low and high depending on how bright the environment is
*/

// input pin for reading changes from LDR
const int analogInPin = A0;
// output pin controlling the motor/fan
const int analogOutPin = 9;
// value read from the LDR
int sensorValue = 0;
// output value of pin 9
int outputValue = 0;
int brightness = 0;    // how bright the LED is
int fadeAmount = 5;    // how many points to fade the LED by

// the setup routine runs once when you press reset:
void setup() {
    // initialize serial communications at 9600 bps
    Serial.begin(9600);
    // initialize pin A0 as input
    pinMode(analogInPin, INPUT);
    // initialize pin 9 as ouput to control motor fan
    pinMode(analogOutPin, OUTPUT);
}

// the loop routine runs over and over again forever:
void loop() {
    // read the analogIn value
    sensorValue = analogRead(analogInPin);
    // map analogInPin to range of values for analogOut from measured values
    outputValue = map(sensorValue, 200, 500, 0, 255);
    // update analogOut value
    analogWrite(analogOutPin, outputValue);
    // print label for sensor value
    Serial.print("brightness = ");
    //print sensor value
    Serial.print(sensorValue);
    //print label for output
    Serial.print("\t output = ");
    //print output value
    Serial.println(outputValue);
    // wait 2 milliseconds before the next loop for the analog-to-digital
    // converter to settle after the last reading:
    delay(2);
}
