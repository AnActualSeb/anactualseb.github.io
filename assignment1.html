// the setup function runs once when you press reset or power the board
void setup() {
// initialize each digital pin used as an output.

// initialize digital pin 9 as an output
pinMode(9, OUTPUT);
// initialize digital pin 10 as an output
pinMode(10, OUTPUT);
// initialize digital pin 11 as an output
pinMode(11, OUTPUT);
}

// the loop function runs over and over again forever
void loop() {
// turns on LEDs from left to right, one by one, then off, using offset delays,

// turn on LED connected to digital pin 9 by setting voltage to HIGH
digitalWrite(9, HIGH);
// wait for 2 seconds
delay(2000);
// turn off LED connected to digital pin 9 by setting voltage to LOW
digitalWrite(9, LOW);
// wait for 2 seconds
delay(2000);
// turn on LED connected to digital pin 10 by setting voltage to HIGH
digitalWrite(10, HIGH);
// wait for 1 second
delay(1000);
// turn off LED connected to digital pin 10 by setting voltage to LOW
digitalWrite(10, LOW);
// wait for 1 second
delay(1000);
// turn on LED connected to digital pin 11 by setting voltage to HIGH
digitalWrite(11, HIGH);
// wait for 500 milliseconds
delay(500);
// turn off LED connected to digital pin 11 by setting voltage to LOW
digitalWrite(11, LOW);
// wait for 500 milliseconds
delay(500);
}
